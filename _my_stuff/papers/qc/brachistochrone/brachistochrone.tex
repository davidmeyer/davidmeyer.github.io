  {%
    \fbox{\scantokens{#1\ignorespaces}}%
 }

% Now make the figure

\begin{figure}[H]
  \centering
  \resizebox{0.65 \textwidth}{!} {    
   \simplebox {                                                                                                                                               % resize figure if you want
     \begin{tikzpicture}
%
% set up some coordinates
%
       \coordinate (A)       at (0,0,0);
       \coordinate (Ay)     at (0,-1.20,0);
       \coordinate (Ax)     at (1.20,0,0);
       \coordinate (Ball)   at (0.5,-0.5,0);
       \coordinate (B)       at (4,-1,0);
       \coordinate (Gy0)   at (3.5,0.25,0);
       \coordinate (Gy1)   at (3.5,-0.5,0);
%
% draw the curve in blue
%
       \draw[blue, thick] (A) to [bend right=50] (B);                                                                                             % draw the curve in blue
       \node[circle, draw=black, fill=red, scale=0.75] at (Ball)  {};                                                                       % draw red ball with a black outline on the curve 
       \fill (A) circle(0.05);                                                                                                                                    % put a dot on the curve at A
       \fill (B) circle(0.05);                                                                                                                                    % put a dot on the curve at B
       \draw[draw=none] (A) node[xshift=-0.30cm, font=\footnotesize, left] {$A \; (x_1,y_1)$};                          % label A
       \draw[draw=none] (B) node[xshift=0.10cm, font=\footnotesize, right] {$B \; (x_2,y_2)$};                         % label B
       \draw[-stealth, thin] (A) -- (Ax) coordinate [label={[font=\scriptsize, above] $x$}];                                    % draw x axis
       \draw[-stealth, thin] (A) -- (Ay) coordinate [label={[font=\scriptsize, left] $y$}];                                         % draw y axis
       \draw[-stealth, thick] (Gy0) -- (Gy1) coordinate [label={[font=\footnotesize, midway, right] $\vec{g}$}];   % draw the gravity vector (g)
      \end{tikzpicture}                                                                                                                                         % end \tikzipcture
      }                                                                                                                                                                  % end \simplebox
    }                                                                                                                                                                    % end \resizebox
  \caption{The Brachistochrone Problem Setup}
  \label{fig:brachistochrone}
