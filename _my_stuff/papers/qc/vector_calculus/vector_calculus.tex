\section{Notation}
Some authors use bold to denote a vector. Adding a "hat"
denotes a unit vector; for example $\mathbf{\hat{i}}$ denotes
the unit vector in the $x$ direction. Using these
definitions the parametric form of a three dimensional 
vector $\mathbf{r}(t)$ would be

\begin{equation*}
\mathbf{r}(t) = g(t) \mathbf{\hat{i}} +h(t) \mathbf{\hat{j}} + k(t) \mathbf{\hat{k}}
\end{equation*}

\bigskip
\noindent
On the other hand, some authors use "over arrow"

\begin{equation*}
\vec{r}(t) = g(t) \vec{i} +h(t) \vec{j} + k(t) \vec{k}
\end{equation*}

\bigskip
\noindent
I've also seen

\begin{equation*}
\vec{r}(t) = g(t) \hat{i} + h(t) \hat{j} + k(t) \hat{k}
\end{equation*}



\bigskip
\noindent
In this document I'll use these forms interchangeably.

\bigskip
\noindent
We will also use $\dot{\mathbf{r}}$ and $\dfrac{d\mathbf{r}}{dt}$
interchangeably.

\section{Arc Length}
Consider a segment of a parametric curve $\mathbf{r} = \mathbf{r}(t)$ 
between two points $P =\mathbf{r}(t)$ and $Q =\mathbf{r}(t + \Delta t)$,
as shown in Figure \ref{fig:delta-r}.

\bigskip
\begin{figure}[H]
\center{\includegraphics[frame,scale=0.40] {images/delta-r.png}}
\caption{$\Delta \mathbf{r} = \mathbf{r}(t + \Delta t) - \mathbf{r}(t)$}
\label{fig:delta-r}
\end{figure}

\bigskip
\noindent
What is the length of the segment of the curve between $P$ and $Q$?
This length is called the arc length and is denoted by $\Delta s$. 
$\Delta s$ can be approximated by the chord length
$\| \Delta \mathbf{r} \|$. Then we can compute the
infinitesimal arc length $ds$ as follows:

\bigskip
\begin{equation*}
\begin{array}{lllll}
\Delta s
&\approx& \| \Delta \mathbf{r} \|							
			&\qquad \qquad \mathrel{\#} \text{approximate $\Delta s$ by $\| \Delta \mathbf{r} \|$} \\
[10 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \|		
			&\qquad \qquad \mathrel{\#} \Delta \mathbf{r}= \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \\
[10 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \| \cdot \dfrac{\Delta t}{\Delta t} 
			&\qquad \qquad \mathrel{\#} \text{multiply by $1 = \dfrac{\Delta t}{\Delta t}$} \\
[10 pt]
&=& \bigg \| \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t 
			&\qquad \qquad \mathrel{\#} \text{assume $\Delta t > 0$} \\
[10 pt]
&\Rightarrow& ds = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \| \cdot dt
			&\qquad \qquad \mathrel{\#} \text{since} {\displaystyle \lim_{\Delta t \to 0} 
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} = \dfrac{d\mathbf{r}}{dt}} \\
\end{array}
\end{equation*}


\bigskip
\bigskip
\noindent
Now we see that

\bigskip
\begin{equation}
\dfrac{ds}{dt} = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \|
\label{eqn:dsdt}
\end{equation}

\noindent
and so

\begin{equation}
\dfrac{ds}{dt} = \| \dot{\mathbf{r}} \| 
\label{eqn:mag_r_dot}
\end{equation}



\bigskip
\noindent
Since $\dfrac{ds}{dt} 
= \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \| 
= \big \| \dot{\mathbf{r}} \big \|$
the unit tangent vector
$\mathbf{T} = \dfrac{\dot{\mathbf{r}}} {\|\dot{\mathbf{r}}\|} 
= \dfrac{d\mathbf{r}}{ds}$.
One way to see this is 



\bigskip
\begin{equation*}
\begin{array}{lllll}
\mathbf{T}
&=& \dfrac{\dot{\mathbf{r}}} {\|\dot{\mathbf{r}}\|} 
			&\hskip 14em \mathrel{\#} \text{definition of the unit tangent vector $\mathbf{T}$}\\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\|\dot{\mathbf{r}} \|} 
			&\hskip 14em \mathrel{\#} \text{definition of $\dot{\mathbf{r}} = \dfrac{d\mathbf{r}}{dt}$} \\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} 
			&\hskip 14em \mathrel{\#} \text{Equation (\ref{eqn:mag_r_dot}): $\|\dot{\mathbf{r}}\| 
			= \dfrac{ds}{dt}$} \\
[25pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} \cdot \dfrac{dt}{dt}
		&\hskip 14em \mathrel{\#} \text{multiply by $1 = \dfrac{dt}{dt}$} \\
[25pt]
&=& \dfrac{d\mathbf{r}}{ds}
			&\hskip 14em \mathrel{\#} \text{simplify}
\end{array}
\end{equation*}
