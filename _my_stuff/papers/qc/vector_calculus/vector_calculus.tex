\section{Notation}
Some authors use bold to denote a vector. Adding a "hat"
denotes a unit vector; for example $\mathbf{\hat{i}}$ denotes
the unit vector in the $x$ direction. Using these
definitions the parametric form of a three dimensional 
vector $\mathbf{r}(t)$ would be

\begin{equation*}
\mathbf{r}(t) = g(t) \, \mathbf{\hat{i}} +h(t) \, \mathbf{\hat{j}} + k(t) \, \mathbf{\hat{k}}
\end{equation*}

\bigskip
\noindent
On the other hand, some authors use "over arrow"

\begin{equation*}
\vec{r}(t) = g(t) \, \vec{i} +h(t) \, \vec{j} + k(t) \, \vec{k}
\end{equation*}

\bigskip
\noindent
I've also seen

\begin{equation*}
\vec{r}(t) = g(t) \, \hat{i} + h(t) \, \hat{j} + k(t) \, \hat{k}
\end{equation*}



\bigskip
\noindent
In this document I'll use these forms interchangeably.

\bigskip
\noindent
We will also use $\dot{\mathbf{r}}$, $\dfrac{d\mathbf{r}}{dt}$
and $\mathbf{r}^{\prime}$ interchangeably.

\section{Arc Length}
Consider a segment of a parametric curve $\mathbf{r}(t)
= g(t) \, \hat{\mathbf{i}} + h(t) \, \hat{\mathbf{j}} + k(t) \, \hat{\mathbf{k}}$
between two points $P =\mathbf{r}(t)$ and $Q =\mathbf{r}(t + \Delta t)$,
as shown in Figure \ref{fig:delta-r}.

\bigskip
\begin{figure}[H]
\center{\includegraphics[frame,scale=0.35] {images/delta-r.png}}
\caption{$\Delta \mathbf{r} = \mathbf{r}(t + \Delta t) - \mathbf{r}(t)$}
\label{fig:delta-r}
\end{figure}

\bigskip
\noindent
What is the length of the segment of the curve between $P$ and $Q$?
This length is called the arc length and is denoted by $\Delta s$. 
$\Delta s$ can be approximated by the chord length
$\| \Delta \mathbf{r} \|$. Then we can compute the
infinitesimal arc length $ds$ as follows:

\smallskip
\begin{equation*}
\begin{array}{lllll}
\Delta s
&\approx& \| \Delta \mathbf{r} \|							
			&\qquad \mathrel{\#} \text{approximate $\Delta s$ by $\| \Delta \mathbf{r} \|$} \\
[12 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \|		
			&\qquad \mathrel{\#} \Delta \mathbf{r}= \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \\
[12 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \| \cdot \dfrac{\Delta t}{\Delta t} 
			&\qquad \mathrel{\#} \text{multiply by $1 = \dfrac{\Delta t}{\Delta t}$} \\
[12 pt]
&=& \bigg \| \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t 
			&\qquad \mathrel{\#} \text{assume $\Delta t > 0$} \\
[12 pt]
&\Rightarrow& \Delta s - \bigg \| \dfrac{\mathbf{r}(t + \Delta t) - 
			\mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t = 0
			&\qquad \mathrel{\#} \text{subtract 
			$\bigg \| \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t$
			from both sides} \\
[15 pt]
&\Rightarrow& {\displaystyle \lim_{\Delta t \to 0} \Bigg [\Delta s - \bigg \| 			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot \Delta t} \Bigg ] 
			= {\displaystyle \lim_{\Delta t \to 0} 0}
			&\qquad \mathrel{\#} \text{take the limit of both sides} \\
[15 pt]
&\Rightarrow& {\displaystyle ds -\lim_{\Delta t \to 0} \Bigg  [ \bigg \| 			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot \Delta t} \Bigg ] 
			= 0
			&\qquad \mathrel{\#} \text{sum rule for limits \cite{properties_of_limits},  
			${\displaystyle \lim_{\Delta t \to 0} 0 = 0}$,
			${\displaystyle \lim_{\Delta t \to 0} \Delta s = ds}$} \\
[15 pt]
&\Rightarrow& {\displaystyle ds = \lim_{\Delta t \to 0} \Bigg [	
			\bigg \|\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot \Delta t} \Bigg ] 
			&\qquad \mathrel{\#} \text{solve for $ds$} \\
[15 pt]
&\Rightarrow& {\displaystyle ds = \lim_{\Delta t \to 0} \bigg \| 			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| 
			\cdot \lim_{\Delta t \to 0} \Delta t}
			&\qquad \mathrel{\#} \text{product rule for limits \cite{properties_of_limits}} \\
[15 pt]
&\Rightarrow& ds = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \| \cdot dt
			&\qquad \mathrel{\#} {\displaystyle \lim_{\Delta t \to 0} 
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} = \dfrac{d\mathbf{r}}{dt}},
			{\displaystyle \lim_{\Delta t \to 0} \Delta t = dt}
\end{array}
\end{equation*}

\bigskip
\noindent
Interestingly we can also see this using the Pythagorean theorem.
Specifically, if we move an infinitesimal distance in the $x$ 
and $y$ directions we get a triangle in the x-y plane
who's sides have length $dx$ and $dy$ and who's hypotenuse 
is $ds$. This scenario is shown for some curve $f(x)$ in 
Figure \ref{fig:pythagorean_theorem}.

\bigskip
\noindent
In the two dimensional case where the parameterization is 
$x = g(t)$ and $y =  h(t)$ we have 
$\mathbf{r}(t) = g(t) \, \hat{\mathbf{i}} + h(t) \, \hat{\mathbf{j}}$.
Further, since derivative is a linear operator 
\cite{wiki:linearity_of_differentiation}
we know that

\bigskip
\begin{equation}
\dfrac{d\mathbf{r}}{dt} = \dfrac{dg}{dt} \, \hat{\mathbf{i}} + \dfrac{dh}{dt} \, \hat{\mathbf{j}}
\label{eqn:drdt}
\end{equation}

\bigskip
\noindent
So in the source domain the Pythagorean theorem 
tells us that $ds = \sqrt{dx^2 + dy^2}$. Using
this expression for $ds$ we can see that



\begin{equation*}
\begin{array}{lllll}
ds
&=& \sqrt{dx^2 + dy^2}				
			&\qquad \qquad \mathrel{\#} \text{by the Pythagorean Theorem (see Figure \ref{fig:pythagorean_theorem})} \\
[15pt]
&=& \sqrt{dg^2 + dh^2}	
			&\qquad \qquad \mathrel{\#} \text{switch to parametric form: $x = g(t)$ and $y = h(t)$} \\
[15pt]
&=& \sqrt{dg^2 + dh^2} \cdot \dfrac{dt}{dt}
			&\qquad \qquad \mathrel{\#} \text{multiply by $1 = \dfrac{dt}{dt}, \; dt > 0$}  \\
[15pt]
&=& \sqrt{dg^2 + dh^2} \cdot \Bigg [ dt \cdot \sqrt{\dfrac{1}{dt^2}} \Bigg ]
			&\qquad \qquad \mathrel{\#} \dfrac{x}{x} = x \cdot \dfrac{1}{\sqrt{x^2}}, \; x > 0  \\
[15pt]
&=& \sqrt{\left (\dfrac{1}{dt^2} \right) \cdot \Big ( dg^2 + dh^2 \Big )} \cdot dt
			&\qquad \qquad \mathrel{\#} \sqrt{x} \cdot \sqrt{y} = \sqrt{x \cdot y} \\
[15pt]
&=& \sqrt{\left (\dfrac{dg}{dt} \right)^2 + \left (\dfrac{dh}{dt} \right)^2} \cdot dt		
			&\qquad \qquad \mathrel{\#} \text{simplify: 
			$\dfrac{1}{x^2} \cdot y^2 = \dfrac{y^2}{x^2} = \left(\dfrac{y}{x} \right )^2$} \\
[15pt]
&=& \sqrt{g^{\prime}(t)^2 + h^{\prime}(t)^2} \cdot dt		
			&\qquad \qquad \mathrel{\#} \text{switch notation: $\dfrac{df}{dt} = f^{\prime}(t)$} \\	
[15pt]
&=& \bigg \| \dfrac{d \mathbf{r}}{dt} \bigg \| \cdot dt
			&\qquad \qquad \mathrel{\#} \text{Equation (\ref{eqn:drdt}) and the definition of $\|\cdot\|$}
\end{array}
\end{equation*}

\noindent
In any event, now we see that

\bigskip
\begin{equation}
\dfrac{ds}{dt} = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \|
\label{eqn:dsdt}
\end{equation}

\noindent
and so

\begin{equation}
\dfrac{ds}{dt} = \| \dot{\mathbf{r}} \| 
\label{eqn:mag_r_dot}
\end{equation}



\bigskip
\setstretch{2.0}
\noindent
Since $\dfrac{ds}{dt} 
= \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \| 
= \big \| \dot{\mathbf{r}} \big \|$ we also 
know that the unit tangent vector 
$\mathbf{T} = \dfrac{\dot{\mathbf{r}}} {\|\dot{\mathbf{r}}\|} 
= \dfrac{d\mathbf{r}}{ds}$ \cite{unit_tangent_and_normal_vectors}.
One way to see this is to notice that

\setstretch{1.0}



\bigskip
\begin{equation*}
\begin{array}{lllll}
\mathbf{T}
&=& \dfrac{\dot{\mathbf{r}}} {\|\dot{\mathbf{r}}\|} 
			&\hskip 10em \mathrel{\#} \text{definition of the unit tangent vector $\mathbf{T}$}\\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\|\dot{\mathbf{r}} \|} 
			&\hskip 10em \mathrel{\#} \text{definition of $\dot{\mathbf{r}} = \dfrac{d\mathbf{r}}{dt}$} \\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} 
			&\hskip 10em \mathrel{\#} \text{Equation (\ref{eqn:mag_r_dot}): $\|\dot{\mathbf{r}}\| 
			= \dfrac{ds}{dt}$} \\
[25pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} \cdot \dfrac{dt}{dt}
		&\hskip 10em \mathrel{\#} \text{multiply by $1 = \dfrac{dt}{dt}$} \\
[25pt]
&=& \dfrac{d\mathbf{r}}{ds}
			&\hskip 10em \mathrel{\#} \text{simplify}
\end{array}
\end{equation*}
