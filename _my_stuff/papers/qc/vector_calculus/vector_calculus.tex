\subsection{Vectors}
Some authors use bold font to denote a vector. Adding a "hat"
denotes a unit vector. A unit vector is defined to be 

\begin{equation*}
\hat{\mathbf{u}} = \dfrac{\mathbf{u}}{\| \mathbf{u} \|}
\end{equation*}

\bigskip
\noindent
that is, $\hat{\mathbf{u}}$ is a vector of unit
length in the $\mathbf{u}$ direction.

\bigskip
\noindent
For example $\mathbf{\hat{i}}$ denotes
the unit vector in the $x$ direction. Using these
definitions the parametric form of a three dimensional 
vector $\mathbf{r}(t)$ would be

\begin{equation*}
\mathbf{r}(t) = g(t) \, \mathbf{\hat{i}} +h(t) \,
\mathbf{\hat{j}} + k(t) \, \mathbf{\hat{k}} 
\end{equation*}

\bigskip
\noindent
In this case $\mathbf{\hat{i}}$, $\mathbf{\hat{j}}$ and  
$\mathbf{\hat{k}}$, the basis vectors in $\mathbb{R}^3$,
have column vector format 

\bigskip
\begin{equation*}
{\displaystyle \mathbf {\hat {i}} 
= {\begin{bmatrix}1\\0\\0\end{bmatrix}},\,\,
\mathbf {\hat {j}} ={\begin{bmatrix}0\\1\\0\end{bmatrix}},\,\,
\mathbf {\hat {k}} ={\begin{bmatrix}0\\0\\1\end{bmatrix}}}
\end{equation*}

\smallskip
\bigskip
\noindent
On the other hand, some authors use "over arrow"

\begin{equation*}
\vec{r}(t) = g(t) \, \vec{i} +h(t) \, \vec{j} + k(t) \, \vec{k} 
\end{equation*}


\noindent
I have also seen

\begin{equation*}
\vec{r}(t) = g(t) \, \hat{i} + h(t) \, \hat{j} + k(t) \, \hat{k}
\end{equation*}

\smallskip
\bigskip
\noindent
In this document I'll use these forms interchangeably. 

\bigskip
\begin{definition}
{\bf Dot Product: } The dot product (or scalar product or inner product) of two
row vectors $\mathbf{a} = (a_1,a_2,\hdots, a_n)$ and $\mathbf{b} = (b_1,b_2,\hdots, b_n)$,
where $\mathbf{a}, \mathbf{b} \in \mathbb{R}^n$, is defined to be the scalar value

\medskip
\begin{equation*}
\mathbf{a} \cdot \mathbf{b} 
= \langle \mathbf{a},\mathbf{b} \rangle 
= \sum\limits_{j = 1}^{n} a_j b_j
= \mathbf{a}\mathbf{b}^{\intercal}
\end{equation*}
\end{definition}

\noindent
According to the Pythagorean theorem, the length of a vector 
$\mathbf{a} = (a_1,a_2,a_3) \in \mathbb{R}^3$, denoted $\|\mathbf{a} \|$,
equals $\sqrt{a_1^2 + a_2^2 + a_3^2}$. The next definition is a 
generalization of the notion of length to vectors of $R^n$.


\bigskip
\begin{definition}
{\bf Euclidean Norm: } The Euclidean norm of a vector

\begin{equation*}
\mathbf{a} = (a_1,a_2,\hdots, a_n) \in \mathbb{R}^{n}
\end{equation*}

\noindent
is defined to be

\begin{equation}
\|\mathbf{a}\| = \sqrt{\langle \mathbf{a}, \mathbf{a}} \rangle = \sqrt{\sum\limits_{j = 1}^{n} a_{j}^{2}}
\label{eqn:magnitude}
\end{equation}
\end{definition}

\begin{theorem} {\bf Cauchy–Schwarz inequality: }

\medskip
\begin{equation*}
\big | \langle \mathbf{a},\mathbf{b} \rangle \big | \leq \| \mathbf{a} \| \| \mathbf{b} \|
\end{equation*}

\bigskip
\noindent
{\bf Proof: } The inequality is trivial if either $
\mathbf{a}$ or $\mathbf{b}$ is zero, so assume that 
neither is. Now, let $\mathbf{x} = \frac{\mathbf{a}}{\|\mathbf{a}\|}$ 
and $\mathbf{y} = \frac{\mathbf{b}}{\|\mathbf{b}\|}$ and so
$\| \mathbf{x} \| = \|\mathbf{y} \| = 1$ ($\mathbf{x} = \mathbf{\hat{a}}$ and
$\mathbf{y} = \mathbf{\hat{b}}$). 

\bigskip
\noindent
One way to prove the Cauchy–Schwarz inequality is to use the 
reverse triangle inequality:

\medskip
\begin{equation}
\big | \|\mathbf{x} \| - \| \mathbf{y} \| \big | \leq \| \mathbf{x} - \mathbf{y} \|
\label{eqn:reverse_triangle_inequality}
\end{equation}


\bigskip
\noindent
We can get the reverse triangle inequality from the triangle inequality 
\cite{vector_norms_and_matrix_norms,wiki:triangle_inequality} by observing 
that

\bigskip
\begin{equation*}
\begin{array}{lllll}
\| \mathbf{x} \|
&=& \| \mathbf{x} + (-\mathbf{y} + \mathbf{y})\| 
					&\quad \mathrel{\#} \text{add 0 to $\mathbf{x}$: }
					\mathbf{x} =  \mathbf{x} + 0 
					= \mathbf{x} + (-\mathbf{y} + \mathbf{y}) \\
[8pt]
&=& \| (\mathbf{x} - \mathbf{y} ) + \mathbf{y} \|
					&\quad \mathrel{\#} \text{addition is associative: 
					$\mathbf{v}_1 + (\mathbf{v}_2 + \mathbf{v}_3) 
					= (\mathbf{v}_1 + \mathbf{v}_2) + \mathbf{v}_3$} \\
[8pt]
&\leq& \| \mathbf{x} - \mathbf{y} \| + \| \mathbf{y} \| 
					&\quad \mathrel{\#} \text{by the triangle inequality:
					$\| \mathbf{v}_1 + \mathbf{v}_2 \| \leq \| \mathbf{v}_1 \| + \| \mathbf{v}_2 \|$} \\
[8pt]
&\Rightarrow& \| \mathbf{x} \| \leq \| \mathbf{x} - \mathbf{y} \| + \| \mathbf{y} \|
					&\quad \mathrel{\#} \text{combine expressions for $\| \mathbf{x} \|$} \\
[8pt]
&\Rightarrow& \| \mathbf{x} \| - \| \mathbf{y} \| \leq \| \mathbf{x} - \mathbf{y} \| 
					&\quad \mathrel{\#} \text{subtract $\| \mathbf{y} \|$ from both sides}\\
[14pt]
\| \mathbf{y} \|
&=& \| \mathbf{y} + (-\mathbf{x} + \mathbf{x})\|
					&\quad \mathrel{\#} \text{add 0 to $\mathbf{y}$: }
					\mathbf{y} = \mathbf{y} + 0
					= \mathbf{y} + (-\mathbf{x} + \mathbf{x}) \\
[8pt]
&=& \| (\mathbf{y} - \mathbf{x}) + \mathbf{x} \|
					&\quad \mathrel{\#} \text{addition is associative} \\
[8pt] 
&\leq& \| \mathbf{y} - \mathbf{x} \| + \| \mathbf{x} \| 
					&\quad \mathrel{\#} \text{by the triangle inequality} \\
[8pt]
&\Rightarrow& \| \mathbf{y} \| \leq \| \mathbf{y} - \mathbf{x} \| + \| \mathbf{x} \|
					&\quad \mathrel{\#} \text{combine expressions for $\| \mathbf{y} \|$}\\
[8pt]
&\Rightarrow& \| \mathbf{y} \| - \| \mathbf{x} \| \leq \| \mathbf{y} - \mathbf{x}\| 
					&\quad \mathrel{\#} \text{subtract $\| \mathbf{x} \|$ from both sides}\\
[8pt]
&\Rightarrow& - \Big [ \| \mathbf{y} \| - \| \mathbf{x} \| \Big ]\geq -\| \mathbf{y} - \mathbf{x}\|
					&\quad \mathrel{\#} \text{multiply both sides by -1: $- 1 * (\mathbf{v}_1 \leq \mathbf{v}_2) 
					\Rightarrow -\mathbf{v}_1 \geq -\mathbf{v}_2$} \\
[8pt]
&\Rightarrow& \| \mathbf{x} \| - \| \mathbf{y} \| \geq -\| \mathbf{x} - \mathbf{y}\|
					&\quad \mathrel{\#} - \| \mathbf{y} - \mathbf{x}\|
					= - \| - ( \mathbf{y} - \mathbf{x} )\| 
					= - \| \mathbf{x} - \mathbf{y} \|
					\text{ by Equation (\ref{eqn:magnitude})} \\
[8pt]
&\Rightarrow& -\| \mathbf{x} - \mathbf{y}\| \leq \| \mathbf{x} \| - \| \mathbf{y} \| 
					&\quad \mathrel{\#} \text{rearrange}


\end{array}
\end{equation*}

\bigskip
\noindent
Combining these two statements gives us the reverse triangle
inequality (Equation (\ref{eqn:reverse_triangle_inequality})):

\medskip
\begin{equation*}
-\| \mathbf{x} - \mathbf{y} \| \leq \| \mathbf{x} \| - \| \mathbf{y} 
\|\leq \| \mathbf{x} - \mathbf{y} \|
\Rightarrow \big | \| \mathbf{x} \| - \| \mathbf{y} \| \big | 
\leq \| \mathbf{x} - \mathbf{y}\|
\end{equation*}


\bigskip
\noindent
Now, using the reverse triangle inequality 
and since $\| \mathbf{x} \| = \| \mathbf{y} \| = 1$ 
(by definition above) we can see that

\begin{equation*}
\begin{array}{lllll}
\big |\|\mathbf{x}\| - \|\mathbf{y}\| \big |
&\leq& \|\mathbf{x} - \mathbf{y} \| 
					&\quad \mathrel{\#} \text{reverse triangle inequality (Equation (\ref{eqn:reverse_triangle_inequality}))} \\
[8pt]
&\Rightarrow& 0 \leq \|\mathbf{x} - \mathbf{y}\|
					&\quad \mathrel{\#} \|\mathbf{x} \| = \| \mathbf{y} \| 
					= 1, \; 1 - 1 = 0, \;  |0| = 0 \\
[8pt]
&\Rightarrow& 0 \leq  \| \mathbf{x} - \mathbf{y} \|^2
					&\quad \mathrel{\#} \|\mathbf{x} - \mathbf{y}\| \geq 0 
					\Rightarrow  \| \mathbf{x} - \mathbf{y} \|^2 \geq 0\\ 
[8pt]
&\Rightarrow& 0 \leq \langle \mathbf{x} - \mathbf{y}, \mathbf{x} - \mathbf{y} \rangle
					&\quad \mathrel{\#} \| \mathbf{v} \|^{2} = \langle \mathbf{v},\mathbf{v} \rangle
					\text{ by Equation (\ref{eqn:magnitude})} \\
[8pt]
&\Rightarrow& {\displaystyle 0 \leq \sum\limits_{j = 1}^{n} (x_j - y_j)^2}
					&\quad \mathrel{\#} \text{by Equation (\ref{eqn:magnitude})} \\
[8pt]
&\Rightarrow& {\displaystyle 0 \leq \sum\limits_{j = 1}^{n} \left ( x_{j}^2 - 2 x_{j} y_{j} + y_{j}^{2} \right )}
					&\quad \mathrel{\#} \text{expand previous line} \\
[8pt]
&\Rightarrow& {\displaystyle 0 \leq \sum\limits_{j = 1}^{n} x_{j}^2 - 2 \sum\limits_{j = 1}^{n} x_{j} y_{j}} 
					+ \sum\limits_{j = 1}^{n} y_{j}^{2}
					&\quad \mathrel{\#} \text{sum is a linear operator} \\
[12pt]
&\Rightarrow& 0 \leq \| \mathbf{x} \|^2 - 2 \langle \mathbf{x},\mathbf{y} \rangle 
					+ \|\mathbf{y}\|^2
					&\quad \mathrel{\#} \text{by Equation (\ref{eqn:magnitude})} \\				
[12pt]
&\Rightarrow& 0 \leq 1 - 2 \langle \mathbf{x},\mathbf{y} \rangle + 1
					&\quad \mathrel{\#} \| \mathbf{x} \| = \|\mathbf{y} \| 
					= 1 \Rightarrow \| \mathbf{x} \|^2 = \|\mathbf{y} \|^2 = 1 \\
[12pt]
&\Rightarrow& 0 \leq 2- 2 \langle \mathbf{x},\mathbf{y} \rangle
					&\quad \mathrel{\#} 1 + 1 = 2 \\
[12pt]
&\Rightarrow& -2 \leq -2 \langle \mathbf{x},\mathbf{y} \rangle
					&\quad \mathrel{\#} \text{subtract 2 from both sides}  \\
[12pt]
&\Rightarrow& 1 \geq \langle \mathbf{x},\mathbf{y} \rangle 
					&\quad \mathrel{\#} \text{divide by -2 and reverse inequality}  \\
[12pt]
&\Rightarrow& \langle \mathbf{x},\mathbf{y} \rangle \leq 1
					&\quad \mathrel{\#} \text{rearrange}  \\
[12pt]
&\Rightarrow& \left \langle \dfrac{\mathbf{a}}{\| \mathbf{a} \|}, 
					{\dfrac{\mathbf{b}}{\|\mathbf{b}\|}} \right\rangle \leq 1
					&\quad \mathrel{\#} \mathbf{x} = \dfrac{\mathbf{a}}{\| \mathbf{a} \|}, \; \mathbf{y} 
					= \dfrac{\mathbf{b}}{\|\mathbf{b}\|} \\
[12pt]
&\Rightarrow& \dfrac{\langle \mathbf{a},\mathbf{b} \rangle}{\|\mathbf{a}\| \|\mathbf{b}\|} \leq 1
					&\quad \mathrel{\#} {\small \text{scalar multiplication rule for dot products:
					$\! \langle c_{1}\mathbf{v}_1, c_{2}\mathbf{v}_2 \rangle 
					= c_{1} c_{2} \langle \mathbf{v}_1, \mathbf{v}_2 \rangle$}}\\
[12pt]
&\Rightarrow& \langle \mathbf{a},\mathbf{b} \rangle \leq \|\mathbf{a}\| \|\mathbf{b} \|
					&\quad \mathrel{\#} \text{multiply both sides by $\|\mathbf{a}\| \|\mathbf{b}\|$}

\end{array}
\end{equation*}


\medskip
\noindent
Next note that the absolute value of $x$, $|x|$,
is defined to be

\medskip
\begin{equation}
|x| = {
  \begin{cases} x, & {\text{if }} x \geq 0 \\
               -x, & {\text{if }} x < 0.
  \end{cases}
}
\label{eqn:absolute_value}
\end{equation}


\bigskip
\noindent
If we replace $\mathbf{a}$ by $-\mathbf{a}$ we see that by 
Equation (\ref{eqn:magnitude}) we have $\|-\mathbf{a}\| 
= \| \mathbf{a}\|$. Further, by the scalar multiplication rule for dot products 
with $c_1 = -1$ and $c_2 = 1$ we have $\langle -\mathbf{a},\mathbf{b} \rangle 
= - \langle \mathbf{a},\mathbf{b} \rangle$. So we can see 
that\footnote{There is nothing special about using $\mathbf{a}$ 
here. We could use $\mathbf{b}$ and get the same result.}

\smallskip
\begin{equation*}
- \langle \mathbf{a},\mathbf{b} \rangle 
\leq \|\mathbf{a}\| \|\mathbf{b} \| \\
\bigskip
\end{equation*}

\bigskip
\setstretch{1.25}
\noindent
One way to see the Cauchy–Schwarz inequality 
is by setting $x$ equal to $\langle \mathbf{a},\mathbf{b} \rangle$ in 
Equation (\ref{eqn:absolute_value}). Then when 
$\langle \mathbf{a},\mathbf{b} \rangle \geq 0$ we have 
$\big | \langle \mathbf{a}, \mathbf{b} \rangle \big | = \langle \mathbf{a},\mathbf{b} \rangle$
and we showed above that $\langle \mathbf{a},\mathbf{b} \rangle \leq \|\mathbf{a}\| \|\mathbf{b} \|$.
Similarily, when $\langle \mathbf{a},\mathbf{b} \rangle < 0$ we have
$\big | \langle \mathbf{a}, \mathbf{b} \rangle \big | = - \langle \mathbf{a},\mathbf{b} \rangle$
and we showed that $- \langle \mathbf{a},\mathbf{b} \rangle \leq \|\mathbf{a}\| \|\mathbf{b} \|$.
Together these give us
$\big | \langle \mathbf{a},\mathbf{b} \rangle \big | \leq \|\mathbf{a}\| \|\mathbf{b} \|$,
the Cauchy–Schwarz inequality. $\square$

\end{theorem}

\subsection{Derivatives}
We will also denote the derivative of a vector $\mathbf{r}$ 
with respect to $t$ by $\dot{\mathbf{r}}$, 
$\dfrac{d\mathbf{r}}{dt}$ or $\mathbf{r}^{\prime}(t)$. 
Note also that by definition

\bigskip
\begin{equation*}
{\displaystyle \dfrac{d\mathbf{r}}{dt} = \lim_{\Delta t \to 0}			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t}}
\end{equation*}

\subsection{Vector Fields}
\label{subsec:vector_fields}
In vector calculus and physics, a vector field is an assignment 
of a vector to each point in a subset of space \cite{galbis2012vector}.
For example, a vector field in the plane can be visualized as 
a collection of arrows with a given magnitude and direction, 
each attached to a point in the plane. Vector fields are often used 
to model the speed and direction of a moving fluid throughout 
space, or the strength and direction of some force, such as the 
magnetic or gravitational force, as it changes from one point to 
another point.

\bigskip
\noindent
For example, the wind velocity vector field for the 2011 
Joplin, MO tornado \cite{joplin_missouri_tornado_2011}
is shown in Figure \ref{fig:wind_velocity_vector_field}.
Here the color represents the wind speed $\| \mathbf{v} \|$, 
where $\mathbf{v}$ is the wind velocity vector.


\bigskip
\bigskip
\begin{figure}[H]
\center{\includegraphics[scale=0.40] 
{images/wind-vector-and-isotachs-representing-a-wind-velocity-field-of-2011-Joplin-MO-tornado.png}}
\caption{2011 Joplin, MO Tornado Wind Velocity Vector Field}
\label{fig:wind_velocity_vector_field}
\end{figure}


\bigskip
\noindent
The general form of a vector field (here in three dimensions) is

\smallskip
\begin{equation*}
\vv{\mathbf{F}}(x,y,z) = P(x,y,z) \, \mathbf{\hat{i}} + Q(x,y,z) \, \mathbf{\hat{j}} + R(x,y,z) \, \mathbf{\hat{k}}
\end{equation*}

\bigskip
\noindent
where $P$, $Q$, and $R$ are scalar functions. That is, they are 
functions $f$ such that $f:\mathbb{R}^3 \to \mathbb{R}$.


\section{Arc Length}
Consider a segment of a parametric curve $\mathbf{r}(t)
= g(t) \, \hat{\mathbf{i}} + h(t) \, \hat{\mathbf{j}} + k(t) \, \hat{\mathbf{k}}$
between two points $P =\mathbf{r}(t)$ and $Q =\mathbf{r}(t + \Delta t)$,
as shown in Figure \ref{fig:delta-r}.

\bigskip
\begin{figure}[H]
\center{\includegraphics[frame,scale=0.35] {images/delta-r.png}}
\caption{$\Delta \mathbf{r} = \mathbf{r}(t + \Delta t) - \mathbf{r}(t)$}
\label{fig:delta-r}
\end{figure}

\bigskip
\noindent
What is the length of the segment of the curve between $P$ and
$Q$?  This length is called the arc length and is denoted by
$\Delta s$.  $\Delta s$ can be approximated by the chord length
$\| \Delta \mathbf{r} \|$. Then we can compute the infinitesimal
arc length $ds$ as follows:

\smallskip
\begin{equation*}
\begin{array}{lllll}
\Delta s
&\approx& \| \Delta \mathbf{r} \|							
			&\qquad \mathrel{\#} \text{approximate $\Delta s$ by $\| \Delta \mathbf{r} \|$} \\
[12 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \|		
			&\qquad \mathrel{\#} \Delta \mathbf{r}= \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \\
[12 pt]
&=& \| \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \| \cdot \dfrac{\Delta t}{\Delta t} 
			&\qquad \mathrel{\#} \text{multiply by $1 = \dfrac{\Delta t}{\Delta t}$} \\
[12 pt]
&=& \bigg \| \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t 
			&\qquad \mathrel{\#} \text{assume $\Delta t > 0$} \\
[12 pt]
&\Rightarrow& \Delta s = \bigg \| \dfrac{\mathbf{r}(t + \Delta t) - 
			\mathbf{r}(t)}{\Delta t} \bigg \| \cdot  \Delta t
			&\qquad \mathrel{\#} \text{combine expressions for $\Delta s$ ($\approx \, \to \, =$ ?)} \\
[15 pt]
&\Rightarrow& {\displaystyle \lim_{\Delta t \to 0} \Delta s =  \lim_{\Delta t \to 0} \Bigg [ 			
			\bigg \| \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot \Delta t} \Bigg ]
			&\qquad \mathrel{\#} \text{take the limit of both sides} \\
[15 pt]
&\Rightarrow& {\displaystyle ds = \lim_{\Delta t \to 0} \Bigg  [ \bigg \| 			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \| \cdot \Delta t} \Bigg ] 
			&\qquad \mathrel{\#} \text{$\Delta t \to 0 \Rightarrow \Delta s \to 0 \Rightarrow \!\!
			{\displaystyle \lim_{\Delta t \to 0} \Delta s = ds}$} \\
[15 pt]
&\Rightarrow& {\displaystyle ds = \lim_{\Delta t \to 0} \bigg \| 			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \|
			\cdot \lim_{\Delta t \to 0} \Delta t}
			&\qquad \mathrel{\#} \text{product rule for limits \cite{properties_of_limits}} \\
[15 pt]
&\Rightarrow& {\displaystyle ds =  \bigg \| \lim_{\Delta t \to 0}			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg \|
			\cdot \lim_{\Delta t \to 0} \Delta t}
			&\qquad \mathrel{\#} \text{for a normed vector space ${\displaystyle X \; \lim \|x_n\| = \|\lim x_n \|}$} \\
[15 pt]
&\Rightarrow& ds = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \| \cdot dt
			&\qquad \mathrel{\#} {\displaystyle \lim_{\Delta t \to 0}			
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} = \dfrac{d\mathbf{r}}{dt},
			\; \lim_{\Delta t \to 0} \Delta t} = dt
\end{array}
\end{equation*}

\bigskip
\noindent
Interestingly we can also see this using the Pythagorean theorem.
Specifically, if we move an infinitesimal distance in the $x$ and
$y$ directions we get a triangle in the x-y plane who's sides
have length $dx$ and $dy$ and who's hypotenuse is $ds$. This
scenario is shown for some curve $f(x)$ in Figure
\ref{fig:pythagorean_theorem}.

\bigskip
\noindent
In the two dimensional case where the parameterization is $x =
g(t)$ and $y = h(t)$ we have $\mathbf{r}(t) = g(t) \,
\hat{\mathbf{i}} + h(t) \, \hat{\mathbf{j}}$.  Further, since
derivative is a linear operator
\cite{wiki:linearity_of_differentiation} we know that

\medskip
\begin{equation}
\dfrac{d\mathbf{r}}{dt} = \dfrac{dg}{dt} \, \hat{\mathbf{i}} + \dfrac{dh}{dt} \, \hat{\mathbf{j}}
\label{eqn:drdt}
\end{equation}

\medskip
\noindent
So in the source domain the Pythagorean theorem 
tells us that $ds = \sqrt{dx^2 + dy^2}$. Using
this expression for $ds$ we can see that



\begin{equation*}
\begin{array}{lllll}
ds
&=& \sqrt{dx^2 + dy^2}				
			&\qquad \qquad \mathrel{\#} \text{Figure \ref{fig:pythagorean_theorem}} \\
[15pt]
&=& \sqrt{dg^2 + dh^2}	
			&\qquad \qquad \mathrel{\#} \text{switch to parametric form: 
			$x = g(t)$ and $y = h(t)$} \\
[15pt]
&=& \sqrt{dg^2 + dh^2} \cdot \dfrac{dt}{dt}
			&\qquad \qquad \mathrel{\#} \text{multiply by 
			$1 = \dfrac{dt}{dt}, \; dt > 0$}  \\
[15pt]
&=& \sqrt{dg^2 + dh^2} \cdot \Bigg [\sqrt{\dfrac{1}{dt^2}} \cdot dt \Bigg ]
			&\qquad \qquad \mathrel{\#} \dfrac{x}{x} 
			= \sqrt{\left (\dfrac{1}{x} \right)^2} \cdot x 
			= \sqrt{\dfrac{1}{x^2}} \cdot x, \; x > 0  \\
[15pt]
&=& \sqrt{\Big ( dg^2 + dh^2 \Big ) \cdot \left (\dfrac{1}{dt^2} \right)} \cdot dt
			&\qquad \qquad \mathrel{\#} \text{simplify: 
			$\sqrt{x} \cdot \sqrt{\dfrac{1}{y}} 
			= \sqrt{x \cdot \dfrac{1}{y}}$} \\
[15pt]
&=& \sqrt{\left (\dfrac{dg}{dt} \right)^2 + \left (\dfrac{dh}{dt} \right)^2} \cdot dt		
			&\qquad \qquad \mathrel{\#} \text{simplify: 
			$\left ( x^2 + y^2 \right ) \cdot \left ( \dfrac{1}{z^2} \right ) 
			= \dfrac{x^2}{z^2} + \dfrac{y^2}{z^2} 
			= \left(\dfrac{x}{z} \right )^2 + \left(\dfrac{y}{z} \right )^2$} \\
[15pt]
&=& \sqrt{g^{\prime}(t)^2 + h^{\prime}(t)^2} \cdot dt		
			&\qquad \qquad \mathrel{\#} \text{switch notation from Leibniz $\to$ Lagrange: 
			$\dfrac{df}{dt} = f^{\prime}(t)$} \\	
[15pt]
&=& \bigg \| \dfrac{d \mathbf{r}}{dt} \bigg \| \cdot dt
			&\qquad \qquad \mathrel{\#} \text{Equation (\ref{eqn:drdt}) 
			and the definition of $\|\cdot\|$}
\end{array}
\end{equation*}

\smallskip
\noindent
In any event, now we see that

\bigskip
\begin{equation}
\dfrac{ds}{dt} = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \|
\label{eqn:dsdt}
\end{equation}

\noindent
and so

\begin{equation}
\dfrac{ds}{dt} = \| \dot{\mathbf{r}} \| 
\label{eqn:mag_r_dot}
\end{equation}


\bigskip
\setstretch{2.0}
\noindent
Since $\dfrac{ds}{dt} = \bigg \| \dfrac{d\mathbf{r}}{dt} \bigg \|
= \big \| \dot{\mathbf{r}} \big \|$ we also know that the unit
tangent vector $\mathbf{T} = \dfrac{\dot{\mathbf{r}}}
{\|\dot{\mathbf{r}}\|} = \dfrac{d\mathbf{r}}{ds}$
\cite{unit_tangent_and_normal_vectors}.  One way to see this is
to notice that

\setstretch{1.0}


\bigskip
\begin{equation*}
\begin{array}{lllll}
\mathbf{T}
&=& \dfrac{\dot{\mathbf{r}}} {\|\dot{\mathbf{r}}\|} 
			&\qquad \mathrel{\#} \text{definition of the unit tangent vector $\mathbf{T}$}\\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\|\dot{\mathbf{r}} \|} 
			&\qquad \mathrel{\#} \text{definition of $\dot{\mathbf{r}} = \dfrac{d\mathbf{r}}{dt}$} \\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} 
			&\qquad \mathrel{\#} \text{Equation (\ref{eqn:mag_r_dot}): $\|\dot{\mathbf{r}}\| 
			= \dfrac{ds}{dt}$} \\
[25pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} \cdot \dfrac{dt}{dt}
		&\qquad\mathrel{\#} \text{multiply by $1 = \dfrac{dt}{dt}$} \\
[25pt]
&=& \dfrac{d\mathbf{r}}{ds}
			&\qquad \mathrel{\#} \text{simplify}
\end{array}
\end{equation*}
