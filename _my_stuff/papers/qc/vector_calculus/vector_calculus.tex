\section{Notation}
Some authors use bold to denote a vector. For example, the parametric form
of a three dimensional vector would be

\begin{equation*}
\mathbf{r}(t) = g(t) \hat{i} +h(t) \hat{j} + k(t) \hat{k}
\end{equation*}

\bigskip
\noindent
On the other hand, some authors use "over arrow"

\begin{equation*}
\vec{r}(t) = g(t) \hat{i} +h(t) \hat{j} + k(t) \hat{k}
\end{equation*}

\bigskip
\noindent
In this document I'll use both forms interchangably.


\section{Arclength}
Consider a segment of a parametric curve $\mathbf{r} = \mathbf{r}(t)$ 
between two points $P \, (\mathbf{r}(t))$ and $Q \, (\mathbf{r}(t + \Delta t))$ 
as shown in Figure \ref{fig:delta-r}.

\bigskip
\begin{figure}[H]
\center{\includegraphics[frame,scale=0.35] {images/delta-r.png}}
\caption{$| \Delta \mathbf{r} | = | \mathbf{r}(t + \Delta t) - \mathbf{r}(t) |$}
\label{fig:delta-r}
\end{figure}

\bigskip
\noindent
What is the length of the segment of the curve between $P$ and $Q$?
This length is called the arclength and is denoted by $\Delta s$. 
$\Delta s$ can be approximated by the chord length
$| \Delta \mathbf{r} | = | \mathbf{r}(t + \Delta t) - \mathbf{r}(t) |$.
So we can note that

\begin{equation*}
\begin{array}{lllll}
\Delta s
&\approx& | \Delta \mathbf{r} |							
			&\qquad \qquad \mathrel{\#} \text{the arclength $\Delta s$ can be approximated by $| \Delta \mathbf{r} |$} \\
[10 pt]
&=& | \mathbf{r}(t + \Delta t) - \mathbf{r}(t) |		
			&\qquad \qquad \mathrel{\#} \Delta \mathbf{r}= \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \\
[10 pt]
&=& \bigg | \mathbf{r}(t + \Delta t) - \mathbf{r}(t) \bigg | \cdot \dfrac{\Delta t}{\Delta t} 
			&\qquad \qquad \mathrel{\#} \text{multiply by $1 = \dfrac{\Delta t}{\Delta t}$} \\
[10 pt]
&=& \bigg | \dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} \bigg |\cdot  \Delta t 
			&\qquad \qquad \mathrel{\#} \text{assume $\Delta t > 0$} \\
[10 pt]
&\Rightarrow& ds = \bigg | \dfrac{d\mathbf{r}}{dt} \bigg | \cdot dt
			&\qquad \qquad \mathrel{\#} {\displaystyle \lim_{\Delta t \to 0} 
			\dfrac{\mathbf{r}(t + \Delta t) - \mathbf{r}(t)}{\Delta t} = \dfrac{d\mathbf{r}}{dt}} \\
\end{array}
\end{equation*}

\bigskip
\noindent
So the infinitesimal arclength $ds$ is

\begin{equation}
ds = \bigg | \dfrac{d\mathbf{r}}{dt} \bigg | \cdot dt 
\label{eqn:arclength}
\end{equation}


\bigskip
\noindent
Now we can see that $\dfrac{ds}{dt} 
= \bigg | \dfrac{d\mathbf{r}}{dt} \bigg | 
= \big | \dot{\mathbf{r}} \big |$
and so the unit tangent vector
$\mathbf{T} = \dfrac{\dot{\mathbf{r}}} {|\dot{\mathbf{r}}|} = \dfrac{d\mathbf{r}}{ds}$,
since


\bigskip
\begin{equation*}
\begin{array}{lllll}
\mathbf{T}
&=& \dfrac{\dot{\mathbf{r}}} {|\dot{\mathbf{r}}|} 
			&\qquad \qquad \qquad \qquad \mathrel{\#} \text{definition of the unit tangent vector $\mathbf{T}$}\\
[15 pt]
&=& \dfrac{\left (\dfrac{d\mathbf{r}}{dt}\right )}{\left (\dfrac{ds}{dt} \right )} 
			&\qquad \qquad \qquad \qquad \mathrel{\#} \text{$\dot{\mathbf{r}} = \dfrac{d\mathbf{r}}{dt}$ and $|\dot{\mathbf{r}}| = \dfrac{ds}{dt}$ } \\
[25pt]
&=& \dfrac{d\mathbf{r}}{ds}
			&\qquad \qquad \qquad \qquad \mathrel{\#} \text{simplify}
\end{array}
\end{equation*}
