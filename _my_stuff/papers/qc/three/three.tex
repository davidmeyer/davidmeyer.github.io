\noindent
Consider:

\begin{center}
\begin{equation*}
\begin{array}{llll}
3 
&=&  \sqrt{9}
				&\qquad \qquad \qquad \mathrel{\#} \sqrt{9} = 3 \\
[8pt]
&=&  \sqrt{1 + 8}
				&\qquad \qquad \qquad \mathrel{\#} 9 = 1 + 8 \\
[8pt]
&=&  \sqrt{1 + 2*4}
				&\qquad \qquad \qquad \mathrel{\#} 8 = 2 * 4 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{16}}
				&\qquad \qquad \qquad \mathrel{\#} \sqrt{16} = 4 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 15}}
				&\qquad \qquad \qquad \mathrel{\#} 16 = 1 + 15 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3*5}}
				&\qquad \qquad \qquad \mathrel{\#} 15 = 3 * 5 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{25}}}
				&\qquad \qquad \qquad \mathrel{\#} \sqrt{25} = 5 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4*6}}}
				&\qquad \qquad \qquad \mathrel{\#} 25 = 1 + 4*6 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4 \sqrt{36}}}}
				&\qquad \qquad \qquad \mathrel{\#} \sqrt{36} = 6  \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4 \sqrt{1 + 35}}}}
				&\qquad \qquad \qquad \mathrel{\#} 36 = 1 + 35 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4 \sqrt{1 + 5*7}}}}
				&\qquad \qquad \qquad \mathrel{\#} 35 = 5 * 7 \\
[8pt]
&=&  \sqrt{1 + 2 \sqrt{1 + 3 \sqrt{1 + 4 \sqrt{1 + 5 \sqrt{49}}}}}
				&\qquad \qquad \qquad \mathrel{\#} \sqrt{49} = 7 \\
[8pt]
&=& \cdots


\end{array}
\end{equation*}
\end{center}

\bigskip
\noindent
and so apparently $3 = \sqrt{1 + 2*\sqrt{1 + 3 \sqrt{1 + 4  \sqrt{1 + 5 \sqrt{1 + \hdots}}}}}$.
