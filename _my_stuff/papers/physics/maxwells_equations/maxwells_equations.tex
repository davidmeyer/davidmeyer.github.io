while $\nabla \times \mathbf{F}$ is the cross product \cite{dmm:vector_calculus}.

\section{Maxwell's Equations (differential form)}
\label{sec:maxwells_equations_differential_form}

\medskip
\begin{enumerate}
\item Gauss' law for electricity
\label{item:gauss_law_electricity}
\begin{equation*}
\begin{array}{llll}
\nabla \cdot \vE 
&=& \dfrac{\rho}{\epsilon_0} 	
			&\hspace{8em} \mathrel{\#} \text{a charge ($\rho$) 
				creates an electric field ($\vE$)$\hspace{7em}$} \\
\end{array}
\end{equation*}

\item Gauss' law for magnetism
\label{item:gauss_law_magnetism}	
\begin{equation*}
\begin{array}{rlll}
\nabla \cdot \vB 
&=& 0							
			&\hspace{8em} \mathrel{\#} \text{magnetic monopoles 
				do not exist $\hspace{10em}$}
\end{array}
\end{equation*}

\item Faraday's law of induction
\label{item:faradays_law_of_induction}
\begin{equation*}
\begin{array}{rlll}
\nabla \times \vE 
&=& -\dfrac{\partial \vB}{\partial t} 	&
			\hspace{6em} \mathrel{\#} \text{a moving magnetic 
				field ($\vB$) creates an electric field ($\vE$)}
\end{array}
\end{equation*}


\item  Ampere's law	(with Maxwell's modification)
\label{item:ampers_law}
\begin{equation*}
\begin{array}{rlll}
\nabla \times \vB 
&=& \mu_0 \vJ + \mu_0 \epsilon_0 \dfrac{\partial \vE}{\partial t}					
			&\hspace{2em} \mathrel{\#} \text{an electric field ($\vE$)
				creates a magnetic field ($\vB$) $\hspace{3em}$}
\end{array}
\end{equation*}
\end{enumerate}
